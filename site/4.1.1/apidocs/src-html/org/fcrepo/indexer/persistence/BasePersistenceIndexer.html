016package org.fcrepo.indexer.persistence;
017
018import static org.apache.commons.lang.StringUtils.substringAfterLast;
019import static org.slf4j.LoggerFactory.getLogger;
020
021import java.io.File;
022import java.io.IOException;
023import java.net.URLEncoder;
024import java.net.URI;
025import java.nio.file.Files;
026import java.nio.file.LinkOption;
027import java.nio.file.Path;
028import java.nio.file.Paths;
029
030import org.apache.commons.lang3.StringUtils;
031import org.fcrepo.indexer.SynchIndexer;
032import org.slf4j.Logger;
033
034/**
035 * Base indexer class to extend for persisting various forms of metadata to disk.
036 * @author ajs6f
037 * @author Esm√© Cowles
038 * @author lsitu
039 * @since 2014-10-20
040**/
041public abstract class BasePersistenceIndexer<Content, File> extends SynchIndexer<Content, File> {
042
043    private static final Logger LOGGER = getLogger(BasePersistenceIndexer.class);
044
045    private final String pathName;
046    private final String extension;
047
048    /**
049     * Constructor.
050     * @param pathName Base directory for persisting files.
051     * @param extension Filename extension to append to file names.
052    **/
053    public BasePersistenceIndexer( final String pathName, final String extension ) {
054        this.pathName = pathName;
055        this.extension = extension;
056    }
057
058    /**
059     * Return the path where a given record should be persisted.
060     * @param id The record's URI
061     * @return the path where a given record should be persisted
062     * @throws IOException if IO exception occurred
063    **/
064    protected Path pathFor(final URI id) throws IOException {
065
066        // strip the http protocol and replace column(:) in front of the port number
067        String fullPath = id.toString().substring(id.toString().indexOf("//") + 2);
068        fullPath = StringUtils.substringBefore(fullPath, "/").replace(":", "/") +
069                "/" + StringUtils.substringAfter(fullPath, "/");
070        // URL encode the id
071        final String idPath = URLEncoder.encode(substringAfterLast(fullPath, "/"), "UTF-8");
072
073        // URL encode and build the file path
074        final String[] pathTokens = StringUtils.substringBeforeLast(fullPath, "/").split("/");
075        final StringBuilder pathBuilder = new StringBuilder();
076        for (final String token : pathTokens) {
077            if (StringUtils.isNotBlank(token)) {
078                pathBuilder.append(URLEncoder.encode(token, "UTF-8") + "/");
079            }
080        }
081
082        fullPath = pathBuilder.substring(0, pathBuilder.length() - 1).toString();
083
084        final Path dir = Paths.get(pathName, fullPath);
085        if (Files.notExists(dir, LinkOption.NOFOLLOW_LINKS)) {
086            Files.createDirectories(Paths.get(pathName, fullPath));
087        }
088        return Paths.get(dir.toString(), idPath + extension);
089    }
090}

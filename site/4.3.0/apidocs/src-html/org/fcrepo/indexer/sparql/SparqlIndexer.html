053/**
054 * Indexes triples from Fedora into a triplestore using SPARQL Update.
055 *
056 * @author Esm√© Cowles
057 * @author ajs6f
058 * @since Aug 19, 2013
059**/
060public class SparqlIndexer extends AsynchIndexer<Model, Void> {
061
062    private String queryBase;
063    private String updateBase;
064    private boolean formUpdates = false;
065
066    private static final Logger LOGGER = getLogger(SparqlIndexer.class);
067
068    /**
069     * Number of threads to use for operating against the triplestore.
070     */
071    private static final Integer THREAD_POOL_SIZE = 5;
072
073    private ListeningExecutorService executorService =
074        listeningDecorator(newFixedThreadPool(THREAD_POOL_SIZE));
075
076    /**
077     * Remove any current triples about the Fedora object and replace them with
078     * the provided content.
079     * {@literal @content RDF in N3 format.}
080    **/
081    @Override
082    public Callable<Void> updateSynch(final URI pid, final Model model) {
083        LOGGER.debug("Received update for: {}", pid);
084        removeSynch(pid, false, true);
085        // build a list of triples
086        final StmtIterator triples = model.listStatements();
087        final QuadDataAcc add = new QuadDataAcc();
088        while ( triples.hasNext() ) {
089            add.addTriple( triples.nextStatement().asTriple() );
090        }
091
092        // send update to server
093        LOGGER.debug("Sending update request for pid: {}", pid);
094        return exec(new UpdateRequest(new UpdateDataInsert(add)));
095    }
096
097    /**
098     * Perform a DESCRIBE query for triples about the Fedora object and remove
099     * all triples with subjects starting with the same subject.
100    **/
101    @Override
102    public Callable<Void> removeSynch(final URI subject) {
103        return removeSynch(subject, true, false);
104    }
105
106    @VisibleForTesting
107    protected Callable<Void> removeSynch(final URI subject, final boolean recursive, final boolean blocking) {
108
109        LOGGER.debug("Received remove for: {}", subject);
110        // find triples/quads to delete
111        final String describeQuery = "DESCRIBE <" + subject.toString() + ">";
112        final QueryEngineHTTP qexec = buildQueryEngineHTTP(describeQuery);
113        final Iterator<Triple> results = qexec.execDescribeTriples();
114
115        // build list of triples to delete
116        final Set<String> uris = new HashSet<>();
117        while ( results.hasNext() ) {
118            final Triple triple = results.next();
119
120            // add subject uri, if it is part of this object
121            if ( triple.getSubject().isURI() ) {
122                final String uri = ((Node_URI)triple.getSubject()).getURI();
123                if ( matches(subject, uri) ) {
124                    uris.add(uri);
125                }
126            }
127
128            if ( recursive ) {
129                // add object uri, if it is part of this object
130                if ( triple.getObject().isURI() ) {
131                    final String uri = ((Node_URI)triple.getObject()).getURI();
132                    if ( matches(subject, uri) ) {
133                        uris.add(uri);
134                    }
135                }
136            }
137        }
138        qexec.close();
139
140        // build update commands
141        final UpdateRequest del = buildUpdateRequest();
142        for (final String uri : uris) {
143            final String cmd = "DELETE WHERE { <" + uri + "> ?p ?o }";
144            LOGGER.debug("Executing: {}", cmd);
145            del.add(cmd);
146        }
147
148        // send updates
149        return exec(del, blocking);
150    }
151
152    /**
153     * Determine whether arg candidate is a sub-URI of arg resource, defined as candidate-URI starting
154     * with resource-URI, plus an option suffix starting with a hash (#) or slash (/)
155     * suffix.
156    **/
157    private boolean matches( final URI resource, final String candidate) {
158        // All triples that will match this logic are ones that:
159        // - have a candidate subject or object that equals the target resource of removal, or
160        // - have a candidate subject or object that is prefixed with the resource of removal
161        //    (therefore catching all children).
162        return resource.toString().equals(candidate) || candidate.startsWith(resource.toString() + "/")
163            || candidate.startsWith(resource.toString() + "#");
164    }
165
166    private Callable<Void> exec(final UpdateRequest update) {
167        return exec(update, false);
168    }
169
170    private Callable<Void> exec(final UpdateRequest update, final boolean blocking) {
171        if (update.getOperations().isEmpty()) {
172            LOGGER.debug("Received empty update/remove operation.");
173            return new Callable<Void>() {
174
175                @Override
176                public Void call() {
177                    return null;
178                }
179            };
180        }
181
182        final Callable<Void> callable = new Callable<Void>() {
183
184            @Override
185            public Void call() {
186
187                if (formUpdates) {
188                    // form updates
189                    final UpdateProcessor proc =
190                        createRemoteForm(update, updateBase);
191                    proc.execute();
192                } else {
193                    // normal SPARQL updates
194                    final UpdateProcessRemote proc =
195                        new UpdateProcessRemote(update, updateBase,
196                                emptyContext);
197                    try {
198                        proc.execute();
199                    } catch (final Exception e) {
200                        LOGGER.error(
201                                "Error executing Sparql update/remove!", e);
202                    }
203                }
204                return null;
205            }
206        };
207
208        if (blocking) {
209            try {
210                callable.call();
211            } catch (Exception e) {
212                LOGGER.error("Error calling Sparql update/remove!, {}", e.getMessage());
213            }
214
215        } else {
216            final ListenableFutureTask<Void> task =
217                ListenableFutureTask.create(callable);
218            task.addListener(new Runnable() {
219
220                @Override
221                public void run() {
222                    LOGGER.debug("Completed Sparql update/removal.");
223                    if (LOGGER.isTraceEnabled()) {
224                        try (
225                            final OutputStream buffer = new ByteArrayOutputStream()) {
226                            final IndentedWriter out = new IndentedWriter(buffer);
227                            update.output(out);
228                            LOGGER.trace("Executed update/remove operation:\n{}",
229                                    buffer.toString());
230                            out.close();
231                        } catch (final IOException e) {
232                            LOGGER.error(
233                                    "Couldn't retrieve execution of update/remove operation!",
234                                    e);
235                        }
236                    }
237                }
238            }, executorService);
239            executorService.submit(task);
240        }
241
242        return callable;
243    }
244
245    @Override
246    public IndexerType getIndexerType() {
247        return RDF;

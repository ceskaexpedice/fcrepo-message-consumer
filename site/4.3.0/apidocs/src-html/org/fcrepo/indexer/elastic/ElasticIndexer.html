016package org.fcrepo.indexer.elastic;
017
018import static com.google.common.util.concurrent.MoreExecutors.listeningDecorator;
019import static java.util.concurrent.Executors.newFixedThreadPool;
020import static org.fcrepo.indexer.Indexer.IndexerType.NAMEDFIELDS;
021import static org.slf4j.LoggerFactory.getLogger;
022
023import java.net.URI;
024import java.util.concurrent.Callable;
025
026import javax.annotation.PostConstruct;
027import javax.inject.Inject;
028
029import org.elasticsearch.action.ActionResponse;
030import org.elasticsearch.client.Client;
031import org.fcrepo.indexer.AsynchIndexer;
032import org.fcrepo.indexer.NamedFields;
033import org.slf4j.Logger;
034
035import com.google.common.util.concurrent.ListeningExecutorService;
036
037/**
038 * {@link org.fcrepo.indexer.Indexer} for Elasticsearch.
039 *
040 * @author ajs6f
041 * @since Dec 14, 2013
042 */
043public class ElasticIndexer extends AsynchIndexer<NamedFields, ActionResponse> {
044
045    @Inject
046    private Client client;
047
048    /**
049     * The Elasticsearch index type to use
050     */
051    private String searchIndexType;
052
053    /**
054     * The Elasticsearch index name to use
055     */
056    private String indexName;
057
058    private static final Logger LOGGER = getLogger(ElasticIndexer.class);
059
060
061    /**
062     * Constructs an index with the name supplied.
063     * @throws InterruptedException if interrupted exception occurred
064     */
065    @PostConstruct
066    public void initIndex() throws InterruptedException {
067        if (!client.admin().indices().prepareExists(getIndexName()).execute()
068                .actionGet().isExists()) {
069            LOGGER.debug("Creating index {}.", getIndexName());
070            client.admin().indices().prepareCreate(getIndexName()).execute()
071                    .actionGet();
072        }
073        client.admin().cluster().prepareHealth().setWaitForGreenStatus()
074                .execute().actionGet();
075    }
076
077    @Override
078    public Callable<ActionResponse> removeSynch(final URI id) {
079        return new Callable<ActionResponse>() {
080
081            @Override
082            public ActionResponse call() {
083                return client.prepareDelete(getIndexName(),
084                        getSearchIndexType(), id.toString()).execute().actionGet();
085            }
086        };
087    }
088
089    @Override
090    public Callable<ActionResponse> updateSynch(final URI id, final NamedFields content) {
091        return new Callable<ActionResponse>() {
092
093            @Override
094            public ActionResponse call() {
095                return client.prepareIndex(indexName, searchIndexType, id.toString())
096                        .execute().actionGet();
097
098            }
099        };
100    }
101
102    @Override
103    public IndexerType getIndexerType() {
104        return NAMEDFIELDS;
105    }
106
107    /**
108     * Number of threads to use for operating against the index.
109     */
110    private static final Integer THREAD_POOL_SIZE = 5;
111
112    private ListeningExecutorService executorService =
113        listeningDecorator(newFixedThreadPool(THREAD_POOL_SIZE));
114
115    @Override
116    public ListeningExecutorService executorService() {
117        return executorService;
118    }
119
120    /**
121     * @param searchIndexType the searchIndexType to set
122     */
123    public void setSearchIndexType(final String searchIndexType) {
124        this.searchIndexType = searchIndexType;
125    }
126
127    /**
128     * @param indexName the indexName to set
129     */
130    public void setIndexName(final String indexName) {
131        this.indexName = indexName;
132    }
133
134
135    /**
136     * @return the searchIndexType
137     */
138    public String getSearchIndexType() {
139        return searchIndexType;
140    }
141
142
143    /**
144     * @return the indexName
145     */
146    public String getIndexName() {
147        return indexName;
148    }
149
150}

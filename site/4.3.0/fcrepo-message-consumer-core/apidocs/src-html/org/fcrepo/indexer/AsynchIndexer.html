016package org.fcrepo.indexer;
017
018import static org.slf4j.LoggerFactory.getLogger;
019
020import java.io.IOException;
021import java.net.URI;
022import java.util.concurrent.Callable;
023
024import org.slf4j.Logger;
025
026import com.google.common.util.concurrent.ListenableFuture;
027import com.google.common.util.concurrent.ListenableFutureTask;
028import com.google.common.util.concurrent.ListeningExecutorService;
029
030/**
031 * An {@link Indexer} that executes its operation asynchronously.
032 *
033 * @author ajs6f
034 * @since Dec 8, 2013
035 *
036 * @param <Content> the type of content to index
037 * @param <Result> the type of response to expect from an operation
038 */
039public abstract class AsynchIndexer<Content, Result> implements
040    Indexer<Content> {
041
042    private static final Logger LOGGER = getLogger(AsynchIndexer.class);
043
044    /**
045     * @return The {@link ListeningExecutorService} to use for operation.
046     */
047    public abstract ListeningExecutorService executorService();
048
049    @Override
050    public ListenableFuture<Result> update(final URI identifier, final Content content) throws IOException {
051        LOGGER.debug("Received update for identifier: {}", identifier);
052
053        final ListenableFutureTask<Result> task =
054            ListenableFutureTask.create(updateSynch(identifier, content));
055        task.addListener(new Runnable() {
056            @Override
057            public void run() {
058                synchronized (this) {
059                    notifyAll();
060                }
061            }
062        }, executorService());
063        executorService().submit(task);
064        LOGGER.debug("Issued task to execution pool for identifier: {}",
065                identifier);
066        return task;
067    }
068
069    @Override
070    public ListenableFuture<Result> remove(final URI identifier) throws IOException {
071        LOGGER.debug("Received remove for identifier: {}", identifier);
072        final ListenableFutureTask<Result> task =
073            ListenableFutureTask.create(removeSynch(identifier));
074        task.addListener(new Runnable() {
075            @Override
076            public void run() {
077                synchronized (this) {
078                    notifyAll();
079                }
080            }
081        }, executorService());
082        executorService().submit(task);
083        return task;
084    }
085
086    /**
087     * @param identifier the URI identifier
088     * @return callable
089     */
090    public abstract Callable<Result> removeSynch(final URI identifier);
091
092    /**
093     * @param identifier the URI identifier
094     * @param content the update content
095     * @return callable
096     */
097    public abstract Callable<Result> updateSynch(final URI identifier, final Content content);
098
099}

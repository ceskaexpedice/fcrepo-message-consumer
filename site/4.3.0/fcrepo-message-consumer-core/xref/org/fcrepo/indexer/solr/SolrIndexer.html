19  import static com.google.common.collect.Maps.transformEntries;
20  import static com.google.common.util.concurrent.MoreExecutors.listeningDecorator;
21  import static java.util.Arrays.asList;
22  import static java.util.concurrent.Executors.newFixedThreadPool;
23  import static org.fcrepo.indexer.Indexer.IndexerType.NAMEDFIELDS;
24  import static org.slf4j.LoggerFactory.getLogger;

26  import java.io.IOException;
27  import java.net.URI;
28  import java.util.Collection;
29  import java.util.Map;
30  import java.util.concurrent.Callable;

32  import org.apache.solr.client.solrj.SolrServer;
33  import org.apache.solr.client.solrj.SolrServerException;
34  import org.apache.solr.client.solrj.response.UpdateResponse;
35  import org.apache.solr.common.SolrInputDocument;
36  import org.apache.solr.common.SolrInputField;
37  import org.fcrepo.indexer.AsynchIndexer;
38  import org.fcrepo.indexer.NamedFields;
39  import org.slf4j.Logger;

42  import com.google.common.collect.Maps.EntryTransformer;

45  /**
46   * A Solr Indexer (stub) implementation that adds some basic information to a
47   * Solr index server.
48   *
49   * @author ajs6f
50   * @author yecao
51   * @since Nov 2013
52   */
53  public class SolrIndexer extends AsynchIndexer<NamedFields, UpdateResponse> {
54  
55      public static final String CONFIGURATION_FOLDER =
56          "fedora:system/fedora:transform/fedora:ldpath/";
57  
58      // TODO make index-time boost somehow adjustable, or something
59      public static final Long INDEX_TIME_BOOST = 1L;
60  
61      private final SolrServer server;
62  
63      /**
64       * Number of threads to use for operating against the index.
65       */
66      private static final Integer THREAD_POOL_SIZE = 5;
67  
68      private ListeningExecutorService executorService =
69          listeningDecorator(newFixedThreadPool(THREAD_POOL_SIZE));
70  
71  
72      private static final Logger LOGGER = getLogger(SolrIndexer.class);
73  
74      /**
75       * solrServer instance is autowired in indexer-core.xml
76       * @param solrServer the solr server
77       */
78      @Autowired
79      public SolrIndexer(final SolrServer solrServer) {
80          this.server = solrServer;
81      }
82  
83      @Override
84      public Callable<UpdateResponse> updateSynch(final URI id, final NamedFields fields) {
85          LOGGER.debug("Received request for update to: {}", id);
86          return new Callable<UpdateResponse>() {
87  
88              @Override
89              public UpdateResponse call() {
90                  try {
91                      LOGGER.debug(
92                              "Executing request to Solr index for identifier: {} with fields: {}",
93                              id, fields);
94                      // add the identifier of the resource as a unique index-key
95                      fields.put("id", asList(id.toString()));
96                      // pack the fields into a Solr input doc
97                      final SolrInputDocument inputDoc = fromMap(fields);
98                      LOGGER.debug("Created SolrInputDocument: {}", inputDoc);
99  
100                     final UpdateResponse resp = server.add(inputDoc);
101                     if (resp.getStatus() == 0) {
102                         LOGGER.debug("Update request was successful for: {}",
103                                 id);
104                     } else {
105                         LOGGER.error(
106                                 "Update request returned error code: {} for identifier: {}",
107                                 resp.getStatus(), id);
108                     }
109                     LOGGER.debug("Received result from Solr request.");
110                     return resp;
111                 } catch (final SolrServerException | IOException e) {
112                     LOGGER.error("Update exception: {}!", e);
113                     throw propagate(e);
114                 }
115             }
116         };
117     }
118 
119     protected SolrInputDocument fromMap(final Map<String, Collection<String>> fields) {
120         LOGGER.debug("Constructing new SolrInputDocument...");
121         return new SolrInputDocument(transformEntries(fields,
122                 collection2solrInputField));
123     }
124 
125     private static EntryTransformer<String, Collection<String>, SolrInputField> collection2solrInputField =
126         new EntryTransformer<String, Collection<String>, SolrInputField>() {
127 
128             @Override
129             public SolrInputField transformEntry(final String key,
130                     final Collection<String> input) {
131                 final SolrInputField field = new SolrInputField(key);
132                 for (final String value : input) {
133                     LOGGER.debug("Adding value: {} to field: {}", value, key);
134                     field.addValue(value, INDEX_TIME_BOOST);
135                 }
136                 return field;
137             }
138         };
139 
140     @Override
141     public Callable<UpdateResponse> removeSynch(final URI uri) {
142         LOGGER.debug("Received request for removal of: {}", uri);
143         return new Callable<UpdateResponse>() {
144 
145             @Override
146             public UpdateResponse call() {
147                 try {
148                     final UpdateResponse resp = server.deleteById(uri.toString());
149                     if (resp.getStatus() == 0) {
150                         LOGGER.debug("Remove request was successful for: {}",
151                                 uri);
152                         server.commit();
153 
154                     } else {
155                         LOGGER.error(
156                                 "Remove request has error, code: {} for uri: {}",
157                                 resp.getStatus(), uri);
158                     }
159                     return resp;
160                 } catch (final SolrServerException | IOException e) {
161                     LOGGER.error("Delete Exception: {}", e);
162                     throw propagate(e);
163                 }
164             }
165         };
166     }
167 
168     @Override
169     public IndexerType getIndexerType() {
170         return NAMEDFIELDS;
171     }
172 
173     @Override
174     public ListeningExecutorService executorService() {
175         return executorService;
176     }
177 
178 
179 }
